# ========== РАБОТА С БАЗОЙ (упрощённо) ==========
def load_user_usage():
    if os.path.exists(USAGE_FILE):
        try:
            with open(USAGE_FILE, 'r') as f:
                return json.load(f)
        except:
            return {}
    return {}


def save_user_usage(usage):
    try:
        with open(USAGE_FILE, 'w') as f:
            json.dump(usage, f)
    except:
        pass


def get_user(user_id):
    usage = load_user_usage()
    user_key = str(user_id)
    today_str = date.today().isoformat()
    if user_key not in usage:
        return "free", 3
    if usage[user_key].get("expires", "") >= today_str:
        return "premium", float('inf')
    return "free", max(0, 3 - usage[user_key].get("count", 0))


def decrease_limit(user_id):
    usage = load_user_usage()
    user_key = str(user_id)
    today_str = date.today().isoformat()
    if user_key not in usage:
        usage[user_key] = {"date": today_str, "count": 1}
    else:
        usage[user_key]["count"] = usage[user_key].get("count", 0) + 1
    save_user_usage(usage)


def set_premium(user_id, days=30):
    usage = load_user_usage()
    user_key = str(user_id)
    from datetime import timedelta
    expires = (date.today() + timedelta(days=days)).isoformat()
    usage[user_key] = {"expires": expires}
    save_user_usage(usage)


# ========== ГЕНЕРАЦИЯ 3D ==========
def generate_meshy_3d(prompt, is_text=True, image_data=None):
    """Создаёт задачу в Meshy и возвращает bytes готовой модели"""
    headers = {
        "Authorization": f"Bearer {MESHY_API_KEY}",
        "Content-Type": "application/json"
    }
    
    if is_text:
        payload = {
            "mode": "async",  # лучше использовать асинхронный режим
            "prompt": prompt,
            "art_style": "realistic",
            "should_remesh": True
        }
        response = requests.post(f"{MESHY_API_URL}/text-to-3d", headers=headers, json=payload, timeout=30)
    else:
        # Отправляем фото как base64 (решает проблемы с доступом)
        payload = {
            "image_base64": image_data,
            "should_remesh": True
        }
        response = requests.post(f"{MESHY_API_URL}/image-to-3d", headers=headers, json=payload, timeout=30)
    
    if response.status_code not in (200, 202):
        raise Exception(f"API error: {response.status_code} - {response.text}")
    
    task_data = response.json()
    task_id = task_data.get("result")
    if not task_id:
        raise Exception("No task_id in response")
    
    # Ожидание завершения
    max_wait = 180
    start_time = time.time()
    endpoint = f"{MESHY_API_URL}/text-to-3d/{task_id}" if is_text else f"{MESHY_API_URL}/image-to-3d/{task_id}"
    
    while time.time() - start_time < max_wait:
        time.sleep(5)
        status_response = requests.get(endpoint, headers=headers, timeout=30)
        if status_response.status_code == 200:
            status_data = status_response.json()
            status = status_data.get("status", "")
            if status == "SUCCEEDED":
                model_urls = status_data.get("model_urls", {})
                model_url = model_urls.get("glb")
                if model_url:
                    model_content = requests.get(model_url, timeout=60)
                    if model_content.status_code == 200:
                        return model_content.content
                    else:
                        raise Exception("Failed to download model")
                else:
                    raise Exception("No model URL")
            elif status == "FAILED":
                raise Exception("Generation failed")
        else:
            print(f"Status check error: {status_response.status_code}")
    
    raise Exception("Timeout waiting for generation")

@dp.message(Form.waiting_text_prompt)
async def generate_from_text(message: Message, state: FSMContext):
    await state.clear()
    sub_tier, gens_left = get_user(message.from_user.id)
    if sub_tier != 'premium' and gens_left <= 0:
        await message.answer("❌ Лимит исчерпан. Загляни в /start -> Профиль.")
        return
    
    await message.answer("⏳ Магия начинается... Генерирую 3D-модель через Meshy AI. Это займёт 1-2 минуты.", parse_mode="HTML")
    
    try:
        model_data = generate_meshy_3d(message.text, is_text=True)
        if sub_tier != 'premium':
            decrease_limit(message.from_user.id)
        await message.answer_document(
            types.BufferedInputFile(model_data, filename="model.glb"),
            caption="✅ Вот твоя 3D-модель!",
            reply_markup=result_menu()
        )
    except Exception as e:
        await message.answer(f"❌ Ошибка генерации: {str(e)}")


@dp.message(Form.waiting_photo)
async def generate_from_photo(message: Message, state: FSMContext):
    await state.clear()
    sub_tier, gens_left = get_user(message.from_user.id)
    if sub_tier != 'premium' and gens_left <= 0:
        await message.answer("❌ Лимит исчерпан. Загляни в /start -> Профиль.")
        return
    
    if not message.photo:
        await message.answer("❌ Пожалуйста, отправь фотографию (JPG или PNG).")
        return
    
    await message.answer("⏳ Обрабатываю фото... Это займёт 1-2 минуты.", parse_mode="HTML")
    
    try:
        # Скачиваем фото из Telegram
        file_id = message.photo[-1].file_id
        file = await bot.get_file(file_id)
        file_data = await bot.download_file(file.file_path)
        
        # Кодируем в base64
        encoded_image = base64.b64encode(file_data.read()).decode('utf-8')
        
        # Генерируем модель
        model_data = generate_meshy_3d("", is_text=False, image_data=encoded_image)
        
        if sub_tier != 'premium':
            decrease_limit(message.from_user.id)
        
        await message.answer_document(
            types.BufferedInputFile(model_data, filename="model.glb"),
            caption="✅ Вот твоя 3D-модель по фото!",
            reply_markup=result_menu()
        )
    except Exception as e:
        await message.answer(f"❌ Ошибка генерации: {str(e)}")
